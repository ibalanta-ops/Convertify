function audioBufferToMp3(audioBuffer) { const channels = audioBuffer.numberOfChannels; const sampleRate = audioBuffer.sampleRate; const totalSamples = audioBuffer.length; const mp3Data = new Uint8Array(totalSamples); // Proper interleaving for stereo if (channels === 2) { // Interleave audio data for stereo for (let i = 0; i < totalSamples; i += 2) { mp3Data[i] = audioBuffer.getChannelData(0)[i / 2]; // Left channel mp3Data[i + 1] = audioBuffer.getChannelData(1)[i / 2]; // Right channel } } else if (channels === 1) { // Single channel audio mp3Data.set(audioBuffer.getChannelData(0)); } // Improved mp3 encoder settings for audio channels handling const encoder = new Mp3Encoder(sampleRate, channels); const mp3File = encoder.encodeBuffer(mp3Data); return mp3File; } // Fix MIME type correction from 'audio/mp3' to 'audio/mpeg' const MIME_TYPE = 'audio/mpeg'; // Improved file validation and error handling function validateAudioFile(file) { const validTypes = ['audio/mpeg', 'audio/wav', 'audio/ogg']; if (!validTypes.includes(file.type)) { throw new Error('Invalid file type!'); } if (file.size > 10485760) { // Limit file size to 10MB throw new Error('File size exceeds the limit!'); } return true; } // Ensure all conversion handlers properly process files function convertAudioFile(file) { try { validateAudioFile(file); // Proceed with conversion logic here } catch (error) { console.error('Error during conversion:', error); } }
